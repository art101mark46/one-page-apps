const video = document.getElementById('webcam'); const canvas = document.getElementById('overlay-canvas'); const snapBtn = document.getElementById('snap-btn'); const exportBtn = document.getElementById('export-btn'); const logBox = document.getElementById('log-box'); const labelInput = document.getElementById('expression-label'); const fileLoader = document.getElementById('file-loader'); const fileCountSpan = document.getElementById('file-count'); const uiEntryList = document.getElementById('ui-entry-list'); const countdownOverlay = document.getElementById('countdown-overlay'); const modal = document.getElementById('preview-modal'); const modalTitle = document.getElementById('modal-title'); const modalImg = document.getElementById('modal-image'); const closeModalBtn = document.getElementById('close-modal-btn'); let latestDetection = null; let datasetSessionMemory = []; function log(msg) { logBox.innerHTML += `
[${new Date().toLocaleTimeString()}] ${msg}
`; logBox.scrollTop = logBox.scrollHeight; } async function init() { const MODEL_URL = 'https://cdn.jsdelivr.net/npm/@vladmandic/face-api/model/'; try { log("Downloading advanced tracking engines..."); await faceapi.nets.tinyFaceDetector.loadFromUri(MODEL_URL); await faceapi.nets.faceLandmark68Net.loadFromUri(MODEL_URL); log("Engines ready. Booting mobile-optimized stream..."); startVideo(); } catch (err) { log(`Init Error: ${err.message}`); } } function startVideo() { navigator.mediaDevices.getUserMedia({ video: { width: { ideal: 1280 }, height: { ideal: 720 }, facingMode: "user" } }) .then(stream => { video.srcObject = stream; video.onloadedmetadata = () => { canvas.width = video.videoWidth; canvas.height = video.videoHeight; snapBtn.disabled = false; processWireframeLoop(); }; }) .catch(err => log(`Webcam Connection Error: ${err.message}`)); } async function processWireframeLoop() { if (video.paused || video.ended) return; const trackingOptions = new faceapi.TinyFaceDetectorOptions({ inputSize: 224, scoreThreshold: 0.5 }); const detection = await faceapi.detectSingleFace(video, trackingOptions).withFaceLandmarks(); const ctx = canvas.getContext('2d'); ctx.clearRect(0, 0, canvas.width, canvas.height); if (detection) { const resizedResults = faceapi.resizeResults(detection, { width: canvas.width, height: canvas.height }); latestDetection = resizedResults; faceapi.draw.drawFaceLandmarks(canvas, resizedResults); } else { latestDetection = null; } requestAnimationFrame(processWireframeLoop); } function refreshDatasetUI() { uiEntryList.innerHTML = ''; fileCountSpan.innerText = datasetSessionMemory.length; if (datasetSessionMemory.length > 0) exportBtn.disabled = false; datasetSessionMemory.forEach((item, index) => { const li = document.createElement('li'); li.className = 'entry-item'; li.innerHTML = `
${item.expressionLabel}
Normalized Map
`; li.addEventListener('click', () => showWireframePreview(index)); uiEntryList.insertBefore(li, uiEntryList.firstChild); }); } function showWireframePreview(index) { const targetRecord = datasetSessionMemory[index]; if (targetRecord) { modalTitle.innerText = `Normalized Template: "${targetRecord.expressionLabel}"`; modalImg.src = targetRecord.wireframeVisual || ""; modal.style.display = "flex"; } } closeModalBtn.addEventListener('click', () => modal.style.display = "none"); fileLoader.addEventListener('change', (event) => { const selectedFile = event.target.files[0]; if (!selectedFile) return; const reader = new FileReader(); reader.onload = function(e) { try { const parsedData = JSON.parse(e.target.result); if (Array.isArray(parsedData)) { datasetSessionMemory = datasetSessionMemory.concat(parsedData); refreshDatasetUI(); } } catch (err) { alert("JSON parse error."); } }; reader.readAsText(selectedFile); }); snapBtn.addEventListener('click', () => { const labelValue = labelInput.value.trim().toLowerCase(); if (!labelValue) return alert("Please declare an expression code text label first."); snapBtn.disabled = true; let secondsRemaining = 2; countdownOverlay.innerText = secondsRemaining; countdownOverlay.style.display = "flex"; const timerInterval = setInterval(() => { secondsRemaining--; if (secondsRemaining > 0) { countdownOverlay.innerText = secondsRemaining; } else { clearInterval(timerInterval); countdownOverlay.style.display = "none"; executeActualCapture(labelValue); snapBtn.disabled = false; } }, 1000); }); function executeActualCapture(labelValue) { if (!latestDetection) return alert("Face track dropped! Try holding steadier."); const box = latestDetection.detection.box; const rawPoints = latestDetection.landmarks.positions; // Convert absolute pixel values into relative percentages inside the face box const normalizedPoints = rawPoints.map(p => ({ x: (p.x - box.x) / box.width, y: (p.y - box.y) / box.height })); const trainingFrame = { expressionLabel: labelValue, landmarksCoordinates: normalizedPoints, wireframeVisual: canvas.toDataURL('image/png'), capturedAt: new Date().toISOString() }; datasetSessionMemory.push(trainingFrame); log(`Logged normalized snapshot for "${labelValue}"`); refreshDatasetUI(); } exportBtn.addEventListener('click', () => { if (datasetSessionMemory.length === 0) return; const dataBlob = new Blob([JSON.stringify(datasetSessionMemory, null, 2)], { type: 'application/json' }); const dummyLink = document.createElement('a'); dummyLink.href = URL.createObjectURL(dataBlob); dummyLink.download = `normalized_expression_dataset.json`; document.body.appendChild(dummyLink); dummyLink.click(); document.body.removeChild(dummyLink); }); init();
