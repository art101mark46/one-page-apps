// ========================================================================= // CONFIGURATION PARAMETERS // ========================================================================= const SUCCESS_REDIRECT_URL = "https://www.google.com"; const DATASET_FILE_URL = "hd_expression_dataset.json"; // ========================================================================= const video = document.getElementById('webcam'); const gameplayDiv = document.getElementById('game-play'); const successScreen = document.getElementById('success-screen'); const promptText = document.getElementById('prompt-text'); const scoldBanner = document.getElementById('scold-banner'); const progressBar = document.getElementById('hold-progress'); let trainedProfiles = []; let randomizedExpressionsQueue = []; let currentChallengeIndex = 0; let mobileOptimizedOptions = null; let matchingFramesCounterCount = 0; const TARGET_HOLD_MS = 5000; const FRAME_INTERVAL_MS = 100; const REQUIRED_MATCHING_FRAMES = TARGET_HOLD_MS / FRAME_INTERVAL_MS; function shuffleArray(array) { for (let i = array.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [array[i], array[j]] = [array[j], array[i]]; } return array; } async function initApplication() { gameplayDiv.style.display = 'block'; promptText.innerHTML = `Downloading dataset matrix...`; try { const response = await fetch(DATASET_FILE_URL); if (!response.ok) throw new Error(`HTTP Error: ${response.status}`); trainedProfiles = await response.json(); const uniqueKeys = [...new Set(trainedProfiles.map(p => p.expressionLabel))]; randomizedExpressionsQueue = shuffleArray(uniqueKeys); promptText.innerHTML = `Initializing tracking engines...`; await initAIModels(); bootWebcam(); } catch (err) { promptText.innerHTML = `Setup Failure:
${err.message}`; } } async function initAIModels() { const MODEL_URL = 'https://cdn.jsdelivr.net/npm/@vladmandic/face-api/model/'; try { await faceapi.nets.tinyFaceDetector.loadFromUri(MODEL_URL); await faceapi.nets.faceLandmark68Net.loadFromUri(MODEL_URL); mobileOptimizedOptions = new faceapi.TinyFaceDetectorOptions({ inputSize: 224, scoreThreshold: 0.5 }); } catch (err) { throw new Error(`Model error: ${err.message}`); } } function bootWebcam() { navigator.mediaDevices.getUserMedia({ video: { width: { ideal: 1280 }, height: { ideal: 720 }, facingMode: "user" } }) .then(stream => { video.srcObject = stream; video.onloadedmetadata = () => { updateChallengePrompt(); executeTrackingLoop(); }; }) .catch(err => alert("Camera configuration rejected.")); } function updateChallengePrompt() { if (currentChallengeIndex < randomizedExpressionsQueue.length) { matchingFramesCounterCount = 0; progressBar.style.width = '0%'; scoldBanner.classList.remove('visible'); const promptLabel = randomizedExpressionsQueue[currentChallengeIndex].toUpperCase().replace('_', ' '); promptText.innerHTML = `VERIFY IDENTITY:
HOLD "${promptLabel}" NOW`; } else { gameplayDiv.style.display = 'none'; successScreen.style.display = 'block'; if (video.srcObject) video.srcObject.getTracks().forEach(t => t.stop()); setTimeout(() => { window.location.href = SUCCESS_REDIRECT_URL; }, 400); } } function generateRoastMessage(target, actual) { return actual === 'unknown' ? `Hold steady and optimize look angle.` : `Expected "${target.replace('_',' ')}", but detected "${actual.replace('_',' ')}"`; } async function executeTrackingLoop() { if (video.paused || video.ended || currentChallengeIndex >= randomizedExpressionsQueue.length) return; const detection = await faceapi.detectSingleFace(video, mobileOptimizedOptions).withFaceLandmarks(); if (detection) { const targetGoalLabel = randomizedExpressionsQueue[currentChallengeIndex]; const matchedExpression = evaluateFaceKNN(detection); if (matchedExpression === targetGoalLabel) { matchingFramesCounterCount++; if (matchingFramesCounterCount > 3) scoldBanner.classList.remove('visible'); progressBar.style.width = `${Math.min((matchingFramesCounterCount / REQUIRED_MATCHING_FRAMES) * 100, 100)}%`; if (matchingFramesCounterCount >= REQUIRED_MATCHING_FRAMES) { currentChallengeIndex++; updateChallengePrompt(); } } else { if (matchingFramesCounterCount > 0) { matchingFramesCounterCount = Math.max(0, matchingFramesCounterCount - 3); progressBar.style.width = `${(matchingFramesCounterCount / REQUIRED_MATCHING_FRAMES) * 100}%`; } scoldBanner.innerText = `⚠️ ${generateRoastMessage(targetGoalLabel, matchedExpression)}`; scoldBanner.classList.add('visible'); } } setTimeout(() => executeTrackingLoop(), FRAME_INTERVAL_MS); } function evaluateFaceKNN(liveDetection) { let closestLabelName = null; let minimumVarianceScore = Infinity; const box = liveDetection.detection.box; const livePoints = liveDetection.landmarks.positions; // Normalize live landmarks to percentages inside the local bounding area box const normalizedLive = livePoints.map(p => ({ x: (p.x - box.x) / box.width, y: (p.y - box.y) / box.height })); for (let profile of trainedProfiles) { let cumulativeDistanceVariance = 0; const baselinePoints = profile.landmarksCoordinates; for (let i = 0; i < 68; i++) { if (!baselinePoints[i] || !normalizedLive[i]) continue; const diffX = baselinePoints[i].x - normalizedLive[i].x; const diffY = baselinePoints[i].y - normalizedLive[i].y; cumulativeDistanceVariance += Math.sqrt(diffX * diffX + diffY * diffY); } if (cumulativeDistanceVariance < minimumVarianceScore) { minimumVarianceScore = cumulativeDistanceVariance; closestLabelName = profile.expressionLabel; } } // Normalized variance threshold limit (since measurements are decimals now, 2.5 is the sweet spot) const MAX_ALLOWED_DISTANCE_VARIANCE_TOLERANCE = 2.5; return (minimumVarianceScore < MAX_ALLOWED_DISTANCE_VARIANCE_TOLERANCE) ? closestLabelName : "unknown"; } initApplication();
